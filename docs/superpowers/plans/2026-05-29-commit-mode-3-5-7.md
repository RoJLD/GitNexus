**Files:** `upstream/gitnexus-web/src/hooks/useAppState.tsx`, `upstream/gitnexus-web/src/components/Timeline.tsx`, `tests/unit/components/Timeline.commits.test.tsx`

- [ ] **1.** `useAppState`: add `compareCommits` (interface + impl + return). Mirrors `enterDiffMode`:
```tsx
  // Compare A↔B between two commits (#3): reconstruct both, diff, show union.
  const compareCommits = useCallback(
    async (shaA: string, shaB: string) => {
      if (!projectName || shaA === shaB) return;
      const baseRepo = projectName.split('@')[0];
      setDiffLoading(true);
      setDiffError(null);
      try {
        preDiffGraphRef.current = graph;
        preDiffProjectRef.current = projectName;
        const fetchAt = async (sha: string) => {
          const r = await fetch(`/graph/at-commit?repo=${encodeURIComponent(baseRepo)}&commit=${encodeURIComponent(sha)}`, { cache: 'no-store' });
          const b = await r.json().catch(() => ({}));
          if (!r.ok) throw new Error(b?.error || `reconstruct ${sha.slice(0,7)} failed (HTTP ${r.status})`);
          return { nodes: b.nodes || [], relationships: b.relationships || [] };
        };
        const [ga, gb] = await Promise.all([fetchAt(shaA), fetchAt(shaB)]);
        const diff = computeGraphDiff(ga, gb);
        const unionGraph = createKnowledgeGraph();
        for (const n of diff.unionNodes) unionGraph.addNode(n);
        for (const e of diff.unionEdges) unionGraph.addRelationship(e);
        setGraph(unionGraph);
        setDiffData(diff);
        setGraphModeState('single');
        setDiffMode({ repoA: `${baseRepo}@${shaA.slice(0,7)}`, repoB: `${baseRepo}@${shaB.slice(0,7)}` });
      } catch (err) {
        setDiffError(err instanceof Error ? err.message : 'Failed to compare commits');
      } finally {
        setDiffLoading(false);
      }
    },
    [projectName, graph, setGraph],
  );
```
Interface: `compareCommits: (shaA: string, shaB: string) => Promise<void>;`. Return: add `compareCommits`. (Exit reuses existing `exitDiffMode`.)

- [ ] **2.** `Timeline.tsx` — destructure `compareCommits`, `diffLoading` (if not already). Add a helper to map a cursor date → nearest windowed commit:
```tsx
  const commitNearest = (iso: string | null) => {
    if (!iso || windowedCommits.length === 0) return null;
    const t = Date.parse(iso);
    let best = windowedCommits[0];
    let bestD = Infinity;
    for (const c of windowedCommits) {
      const d = Math.abs(Date.parse(c.date) - t);
      if (d < bestD) { bestD = d; best = c; }
    }
    return best;
  };
```
